        if A[i].field1 = x then
            idx <- i

Mencari data dengan binary search (syarat data harus diurut terlebih dahulu):
function binarySearch(A : contoharraydata, n : integer, x : integer) -> integer

kamus
    left, right, mid, found : integer

algoritma

    left <- 0
    right <- n - 1
    found <- -1

    while left <= right and found = -1 do

        mid <- (left + right) div 2

        if x < A[mid].field3 then

            right <- mid - 1

        else if x > A[mid].field3 then

            left <- mid + 1

        else

            found <- mid

        endif

    endwhile

    return found

endfunction

Mengurutkan data dengan Selection Sort Ascending (Kecil → Besar):
procedure selectionSortAsc(in/out A : contoharraydata, n : integer)

kamus
    pass, idx, j : integer
    temp : contohdata

algoritma

    for pass <- 1 to n-1 do

        idx <- pass - 1

        for j <- pass to n-1 do

            if A[idx].field3 > A[j].field3 then
                idx <- j
            endif

        endfor

        temp <- A[idx]
        A[idx] <- A[pass-1]
        A[pass-1] <- temp

    endfor

endprocedure

Mengurutkan data dengan Insertion Sort Ascending (Kecil → Besar):
procedure insertionSortAsc(in/out A : contoharraydata, n : integer)

kamus
    pass, j : integer
    temp : contohdata

algoritma

    for pass <- 1 to n-1 do

        temp <- A[pass]
        j <- pass

        while j > 0 and temp.field3 < A[j-1].field3 do

            A[j] <- A[j-1]
            j <- j - 1

        endwhile

        A[j] <- temp

    endfor

endprocedure

Input data dengan sentinet:
procedure bacaData(in/out A : contoharraydata,
                   in/out n : integer)

kamus
    stop : string

algoritma

    n <- 0

    input(stop)

    while stop != "STOP" and n < NMAX do

        A[n].field1 <- stop

        input(A[n].field2)
        input(A[n].field3)
        input(A[n].field4)

        n <- n + 1

        input(stop)

    endwhile

endprocedure
